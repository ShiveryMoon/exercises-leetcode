class Solution:
    def intersection(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: List[int]
        """
        if len(nums1) and len(nums2):
            s1=set(nums1)
            s2=set(nums2)
            return list(s1&s2)
        else:
            return []
