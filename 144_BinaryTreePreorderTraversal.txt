class Solution:
    def preorderTraversal(self, root):
        """
        :type root: TreeNode
        :rtype: List[int]
        """
        l=[]
        self._helper(root,l)
        return l
    
    def _helper(self,root,l):
        if root:
            l.append(root.val)
            self._helper(root.left,l)
            self._helper(root.right,l)
