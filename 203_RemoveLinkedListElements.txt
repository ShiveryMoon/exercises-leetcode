class Solution:
    def removeElements(self, head, val):
        """
        :type head: ListNode
        :type val: int
        :rtype: ListNode
        """
        h=cur=ListNode(0)
        cur.next=head
        while cur.next:
            if cur.next.val == val:
                cur.next=cur.next.next
            else:
                cur=cur.next
        return h.next
