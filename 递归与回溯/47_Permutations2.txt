class Solution:
    def permuteUnique(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        rl=[]
        self.backTracking(nums,[],rl)
        return rl
    
    def backTracking(self,nums,tl,rl):
        if len(nums)==0:
            rl.append(tl)
            return True
        appeared=[]
        for i,num in enumerate(nums):
            if num in appeared:
                continue
            appeared.append(num)
            sl=tl+[num]
            self.backTracking(nums[:i]+nums[i+1:],sl,rl)
