class Solution(object):
    def isValid(self, s):
        """
        :type s: str
        :rtype: bool
        """
        stack=[]
        d={'{':'}','[':']','(':')'}
        for c in s:
            if c in d:
                stack.append(c)
            else:
                if len(stack)!=0 and d[stack.pop()]==c:
                    continue
                else:
                    return False
        return True and not bool(len(stack))
